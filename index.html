public static List buscarTodosLugares(){
List paradises = new ArrayList ();
try {
Connection connection = ParadisesConnection.getConnection();
PreparedStatement statement = connection.prepareStatement(SQL_BUSCAR_TODOS);
ResultSet resultSet = statement.executeQuery();
while (resultSet.next()) {
Paradise paradise = new Paradise(resultSet.getInt("ID"), resultSet.getString("NOME"));
paradises.add(paradise);
}
statement.close();
resultSet.close();
ParadisesConnection.closeConnection();
} catch (SQLException e) {
// TODO Auto-generated catch block
e.printStackTrace();
}
return paradises;
}
